1. GROUP BY department:
SELECT department, AVG(age) AS avg_age
FROM employees
GROUP BY department;

2. WHERE any condition:
SELECT *
FROM employees
WHERE department = 'IT' AND age > 25;

3. HAVING:
SELECT department, AVG(age) AS avg_age
FROM employees
GROUP BY department
HAVING AVG(age) > 30;

4. ORDER BY employee number:
SELECT *
FROM employees
ORDER BY employee_number ASC;
